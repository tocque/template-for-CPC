#常数优化

##读入优化

```c++
struct fastin {
    fastin operator >> (int &i) {
        i = 0; int f = 1; char c;
        do {c = getchar(); if (c == '-') f = -1;} while (c < '0' || c > '9');
        while (c >= '0' && c <= '9') i = i * 10 + c - '0', c = getchar();
        i *= f; return *this;
    }
} fin;
```

##输出优化

```c++
struct fastout {
    const int N = 1180000;
    char buf[N], *p = buf, pool[10], *tail;
    fastout operator << (int x) {
        if (!x) *p++ = '0';
        else {
            if (x < 0) *p++ = '-', x = -x;
            tail = pool;
            while (x) *tail++ = x % 10 + '0', x /= 10;
            while (tail-- != pool) *p++ = *tail;
        }
        return *this;
    }
    fastout operator << (char x) {
        *p++ = c; return *this;
    }

} fout;
```

##优先队列预申请空间

```c++
template <typename T>
struct prque {
    static const int N = 1e5;
    T dat[N]; int cnt = 0;
    void push(dat_t i) { dat[cnt++] = i, push_heap(dat, dat+cnt); }
    T pop() { pop_heap(dat, dat+cnt); return dat[--cnt]; }
    T top() { return dat[0]; }
    bool empty() { return !cnt; }
};
```

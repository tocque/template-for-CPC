```c++
struct disju {
    int pa[N], siz;
    void init(int l) {
        siz = l; rep(i, siz) pa[i] = i;
    }
    inline int rt(int i) {
        int n = i;
        while (n != pa[n]) n = pa[n];
        while (i != pa[i]) i = pa[i], pa[i] = n;
        return n;
    }
    bool isjoint(int a, int b) { return rt(a) == rt(b); }
    void join(int a, int b) {
        pa[rt(b)] = rt(a);
    }
    int blockcnt() {
        int cnt = 0;
        rep (i, siz) {
            if (pa[i] == i) cnt++;
        }
        return cnt;
    }
};
```

####假线段树(单点/区间)

```c++
template<typename T, int N>
struct segtree {
    T dat[N]; int siz;

    void init(int n) {
        siz = 1 << int(ceil(log2(n)));
        memset(dat, 0, sizeof dat);
    }
    // 单点查询-区间修改
    T query(int n) {
        T val = 0; int p = siz + n;
        do {
            stat(val, dat[p]);
        } while (p >>= 1);
        return val;
    }

    void change(int l, int r, T dt) {
        l += siz, r += siz;
        while(l <= r) {
            if (l & 1) upd(l++, dt);
            if (!(r & 1)) upd(r--, dt);
            l >>= 1, r >>= 1;
        }
    }

    // 单点修改-区间查询
    void change(int n, int dt) {
        int p = siz + n;
        dat[p] = dt;
        while (p >>= 1) {
            upd(p);
        }
    }

    void query(int l, int r) {
        T val;
        l += siz, r += siz;
        while(l <= r) {
            if (l & 1) stat(val, l++);
            if (!(r & 1)) stat(val, r--);
            l >>= 1, r >>= 1;
        }
    }
};
```
